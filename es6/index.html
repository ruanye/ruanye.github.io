# 三、Proxy

//一)、 proxy 和 defineProperty的区别
1. proxy 既能代理对象也能代理数组
2. 兼容性 proxy ie11以上  defineProperty 兼容ie8以上
3. proxy是代理，defineProperty 是劫持
4. 配合Reflect,Reflect有13种方法，处理数据更灵活
// 二）、基本用法
let obj = { name: 'lilei' };
let proxy = new Proxy(obj, {
  // target目标对象，key 目标对象的key
  get(target, key) {
    return target[key];
  },
  //val 需要设置的值
  set(target, key, val) {
    target[key] = val;
  }
});
proxy.name = 1;
// 三）、代理数据 实现更新
let obj = {
  name: 'lilei',
  hobit: {
    one: 'football',
    two: 'eat'
  },
	ary:[1,2,3,4]
};
function update() {
  console.log('更新');
}
handle = {
  // target目标对象，key 目标对象的key
  get(target, key) {
    //判断target[key]是不是对象，如果是在返回劫持的对象
    if (typeof target[key] == 'object' && typeof target[key] != null) {
      return new Proxy(target[key], handle);
    }
    return Reflect.get(target, key);
  },
  //val 需要设置的值
  set(target, key, val) {
    // 因为数组的长度改变也会触发更新，处理如下
    if (key == 'length') return true;
    update();
    return Reflect.set(target, key, val);
  }
};
let proxy = new Proxy(obj, handle);
proxy.hobit.one = 'music';
proxy.ary.push(5)
proxy;

# 四、Reflect 的 13 种用法

将 Object 对象的一些明显属于语言内部的方法（比如 Object.defineProperty），放到 Reflect 对象上。现阶段，某些方法同时在 Object 和 Reflect 对象上部署，未来的新方法将只部署在 Reflect 对象上。也就是说，从 Reflect 对象上可以拿到语言内部的方法。

// 里面有部分的对象的方法 放到 Reflect 功能基本一致
// Proxy 中能代理的方法 Reflect 都可以实现
// 1、2 get和set
const obj = {};
Reflect.set(obj, 'name', 'C'); // obj.name = C;
console.log(Reflect.get(obj, 'name'));
//3、 object.definProperty Reflect.definProperty
//Reflect.defineProperty 会有返回值(true/false)表示有没有配置成功
let obj1 = Object.freeze({ name: 1 });
let flag = Reflect.defineProperty(obj1, 'name', {
  value: 100
});
console.log(obj1, flag);
//4、has
let obj2 = { a: 1 };
console.log('a' in obj2);
console.log(Reflect.has(obj2, 'a'));
// 5)getOwnPropertyDescriptor 获取自己私有的属性描述器
let obj3 = { a: 1 };
console.log(Object.getOwnPropertyDescriptor(obj3, 'a'));
console.log(Reflect.getOwnPropertyDescriptor(obj3, 'a'));
//6 ownKeys
let obj4 = {
  a: 1,
  [Symbol()]: 2
};
console.log(Object.keys(obj4));
console.log(Object.getOwnPropertySymbols(obj4));
console.log(Object.getOwnPropertyNames(obj4));
console.log(Reflect.ownKeys(obj4));
// 7)setPrototypeOf 设置原型链
function f() {}
f.prototype.eat = function() {
  console.log('eat');
};
function fn() {}
//fn.__proto__ = f.prototype;
Reflect.setPrototypeOf(fn, f.prototype);
//Object.setPrototypeOf(fn, f.prototype);
//fn = Object.create(f.prototype, { constructor: { value: fn } });
console.log(fn.__proto__ === f.prototype);
fn.eat();
//8、 getPrototypeOf // 获取原型
function FancyThing() {}
const myObj = new FancyThing();
// 旧写法
console.log(Object.getPrototypeOf(myObj));

// 新写法
Reflect.getPrototypeOf(myObj) === FancyThing.prototype;
Object.getPrototypeOf(1);

# 五、Generator

// 基本写法
function* helloWorldGenerator() {
  yield 'hello'; //yield 产出
  yield 'world';
  // return  undefined
}
// 用法 调用 Generator 函数后，该函数并不执行，返回的也不是函数运行结果，而是一个指向内部状态的指针对象
let hw = helloWorldGenerator();
// 调用 遍历器（hw）的next方法,指针指向下一个状态，碰到yield或者return会暂停
hw.next(); //result  {value:'hello',done:false}
hw.next(); //result {value:'world',done:false}
// 如果return后面没有值等于默认返回了一个undefined
hw.next(); //result {value:'undefined',done:false}

//generator函数具有symbol.iterator接口 所以可以使用for of 和 ... 拓展运算符
function* fn() {
  yield 1;
  yield 2;
  yield 3;
}
console.log([...fn()]);
for (let key of fn()) {
  console.log(key);
}
console.log(Array.from(fn()));
// 参数问题 yield表达式的返回值undefined。next 第一次传参无效，第二次传参会传给第一次的yield表达式，以此类推
function* read() {
  // 碰到yield会暂停 所以就等有let a
  let a = yield 1;
  console.log(a);
  let b = yield 2;
  console.log(b);
}
// throw方法抛出的错误要被内部捕获，前提是必须至少执行过一次next方法
// throw()、return() throw 就等于代码里加了throw new Error() return() 就等代码里面加了一句return xxx
let it = read();
it.next();
console.log(it.return(1));
// 回调 -> promise ->generator-> async await
// genrator 和promise 一起使用解决异步问题
function getProvince() {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve('北京');
    }, 2000);
  });
}
function getCity(province) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (province == '北京') {
        resolve('昌平');
      }
    }, 1000);
  });
}
function* getArea() {
  let province = yield getProvince();
  let city = yield getCity(province);
  return province + city;
}
// generator 配合promise的基本使用
let gt = getArea();
let { value } = gt.next();
value.then(data => {
  let { value } = gt.next(data);
  value.then(data => {
    let { value } = gt.next(data);
    console.log(value);
  });
});

// co库的实现原理
function co(it) {
  return new Promise((resolve, reject) => {
    function next(data) {
      // ->next()
      let { value, done } = it.next(data);
      if (!done) {
        Promise.resolve(value).then(
          data => {
            next(data); //- 1 next
          },
          err => {
            it.throw(err);
          }
        );
      } else {
        console.log(value);
        // 如果done为true 表示generator执行完毕
        resolve(value);
      }
    }
    next();
  });
}
co(getArea())
  .then(data => {
    console.log(data);
  })
  .catch(err => {
    console.log(err);
  });

# 七、Decorator 装饰器

装饰器最终需要返还一个函数

@log1()
@log2()
class MyClass {}
function log1() {
  console.log('outer1');
  return function(target) {
    // 装饰类指代的是当前类本身
    console.log('inner1');
  };
}
function log2() {
  console.log('outer2');
  return function() {
    console.log('inner2');
  };
}

# 2.)修饰类中原型上的方法

  • class MyClass {
  @enumerable(false) // 是否可枚举
  getName() {
    return 'hello';
  }
}
function enumerable(boolean) {
  return function(target, key, descriptor) {
    //value.enumerable = true;
    return {
      ...descriptor,
      enumerable: boolean
    };
  };
}

# 2.)修饰类中实例的属性

class MyClass {
  @readonly PI = 3.14;
}

function readonly(target, key, value) {
  value.writable = false;
}
let my = new MyClass();
my.PI = 3.15;
console.log(my.PI);
